6. BUSINESS RULES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  [R1] CreateOrder: Trừ stock + history null→pending
  [R2] confirmed→processing: Bắt buộc Shipment + QC
  [R3] processing→shipping: Bắt buộc trackingCode
  [R4] shipping→delivered: COD→auto paid; set actualDelivery
  [R5] Hủy: Bắt buộc reason + hoàn stock + history
  [R6] Mọi thay đổi → INSERT order_status_history
